import {
	ChatInputCommandInteraction,
	EmbedBuilder,
	GuildMember,
	SlashCommandBuilder,
} from 'discord.js';
import { player } from '../../../index.js';

export default {
	data: new SlashCommandBuilder().setName('queue').setDescription('Displays the queue'),

	async execute(interaction: ChatInputCommandInteraction) {
		const queue = player.getQueue(interaction.guild!);
		const currentTrack = queue.current;
		const tracks = queue.tracks
			.slice(0, 10)
			.map((m, i) => `${i + 1}. **${m.title}** ([link](${m.url}))`);
		const description = `${tracks.join('\n')}${
			queue.tracks.length > tracks.length
				? `\n...${
					queue.tracks.length - tracks.length === 1
						? `${queue.tracks.length - tracks.length} more track`
						: `${queue.tracks.length - tracks.length} more tracks`
				  }`
				: ''
		}`;
		const embed = new EmbedBuilder()
			.setColor('#5864f1')
			.setDescription(description)
			.setFields([
				{ name: 'Now Playing', value: `üé∂ | **${currentTrack.title}** ([link](${currentTrack.url}))` },
			])
			.setTitle('Queue');

		if (!(interaction.member as GuildMember).voice.channel)
			return await interaction.reply({ content: '‚ùå | You are not in a voice channel!' });
		if (!queue || !queue.playing)
			return await interaction.reply({ content: '‚ùå | No music is playing!' });
	},
};
